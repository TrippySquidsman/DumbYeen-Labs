# Windows doesn't like default CephFS Samba permissions
# Use force user and force group
# CephFS permissions are simpler to manage if all operations from Samba use a single user/group:

Host samba.conf:

[cephfs]
   comment = Private Folder
   path = /mnt/pve/cephfs
   writable = yes
   guest ok = no
   valid users = @smbshare
   force user = root
   force group = root
   create mask = 0660
   directory mask = 0770
